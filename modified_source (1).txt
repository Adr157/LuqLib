-- Biblioteca LuqLib
local LuqLib = {}
LuqLib.Elements = {}
LuqLib.Connections = {}

-- Tema padrão
LuqLib.Theme = {
    Background = Color3.fromRGB(25, 25, 25),
    Foreground = Color3.fromRGB(32, 32, 32),
    Accent = Color3.fromRGB(0, 162, 255),
    Text = Color3.fromRGB(255, 255, 255),
    SubText = Color3.fromRGB(200, 200, 200)
}

-- Função para criar uma Janela
function LuqLib:MakeWindow(config)
    config = config or {}
    local Window = {}
    Window.Name = config.Name or "LuqLib Window"
    Window.Tabs = {}

    -- Criando ScreenGui
    local ScreenGui = Instance.new("ScreenGui")
    ScreenGui.Name = "LuqLib"
    ScreenGui.Parent = game:GetService("CoreGui")

    -- Criando Frame Principal
    local MainFrame = Instance.new("Frame")
    MainFrame.Size = UDim2.new(0, 600, 0, 400)
    MainFrame.Position = UDim2.new(0.5, -300, 0.5, -200)
    MainFrame.BackgroundColor3 = LuqLib.Theme.Background
    MainFrame.Parent = ScreenGui

    -- Barra Superior
    local TopBar = Instance.new("Frame")
    TopBar.Size = UDim2.new(1, 0, 0, 40)
    TopBar.BackgroundColor3 = LuqLib.Theme.Accent
    TopBar.Parent = MainFrame

    local Title = Instance.new("TextLabel")
    Title.Text = Window.Name
    Title.Size = UDim2.new(1, 0, 1, 0)
    Title.BackgroundTransparency = 1
    Title.TextColor3 = LuqLib.Theme.Text
    Title.Font = Enum.Font.GothamBold
    Title.TextSize = 18
    Title.Parent = TopBar

    -- Abas
    local TabFrame = Instance.new("Frame")
    TabFrame.Size = UDim2.new(0, 150, 1, -40)
    TabFrame.Position = UDim2.new(0, 0, 0, 40)
    TabFrame.BackgroundColor3 = LuqLib.Theme.Foreground
    TabFrame.Parent = MainFrame

    local ContentFrame = Instance.new("Frame")
    ContentFrame.Size = UDim2.new(1, -150, 1, -40)
    ContentFrame.Position = UDim2.new(0, 150, 0, 40)
    ContentFrame.BackgroundColor3 = LuqLib.Theme.Background
    ContentFrame.Parent = MainFrame

    -- Função para criar Abas
    function Window:MakeTab(tabConfig)
        tabConfig = tabConfig or {}
        local Tab = {}
        Tab.Name = tabConfig.Name or "Nova Aba"
        Tab.Elements = {}

        local TabButton = Instance.new("TextButton")
        TabButton.Text = Tab.Name
        TabButton.Size = UDim2.new(1, 0, 0, 40)
        TabButton.BackgroundColor3 = LuqLib.Theme.Foreground
        TabButton.TextColor3 = LuqLib.Theme.Text
        TabButton.Font = Enum.Font.Gotham
        TabButton.TextSize = 14
        TabButton.Parent = TabFrame

        -- Frame de Conteúdo da Aba
        local TabContent = Instance.new("Frame")
        TabContent.Size = UDim2.new(1, 0, 1, 0)
        TabContent.Visible = false
        TabContent.Parent = ContentFrame

        -- Alternar Visibilidade
        TabButton.MouseButton1Click:Connect(function()
            for _, frame in pairs(ContentFrame:GetChildren()) do
                frame.Visible = false
            end
            TabContent.Visible = true
        end)

        -- Adicionar Elementos à Aba
        function Tab:AddButton(buttonConfig)
            buttonConfig = buttonConfig or {}
            local Button = Instance.new("TextButton")
            Button.Text = buttonConfig.Name or "Botão"
            Button.Size = UDim2.new(0, 200, 0, 50)
            Button.BackgroundColor3 = LuqLib.Theme.Accent
            Button.TextColor3 = LuqLib.Theme.Text
            Button.Font = Enum.Font.GothamBold
            Button.TextSize = 14
            Button.Parent = TabContent
            if buttonConfig.Callback then
                Button.MouseButton1Click:Connect(buttonConfig.Callback)
            end
        end

        -- Outros elementos podem ser adicionados aqui (Toggles, Sliders, etc.)

        table.insert(Window.Tabs, Tab)
        return Tab
    end

    return Window
end

-- Função para aplicar temas personalizados
function LuqLib:ApplyTheme(themeConfig)
    for key, value in pairs(themeConfig) do
        if LuqLib.Theme[key] then
            LuqLib.Theme[key] = value
        end
    end
end

-- Função para criar notificações
function LuqLib:MakeNotification(config)
    config = config or {}
    local Notification = Instance.new("ScreenGui")
    Notification.Name = "Notification"
    Notification.Parent = game:GetService("CoreGui")

    local Frame = Instance.new("Frame")
    Frame.Size = UDim2.new(0, 300, 0, 100)
    Frame.Position = UDim2.new(0.5, -150, 0.5, -50)
    Frame.BackgroundColor3 = LuqLib.Theme.Foreground
    Frame.Parent = Notification

    local Title = Instance.new("TextLabel")
    Title.Text = config.Name or "Notificação"
    Title.Size = UDim2.new(1, 0, 0.4, 0)
    Title.BackgroundTransparency = 1
    Title.TextColor3 = LuqLib.Theme.Text
    Title.Font = Enum.Font.GothamBold
    Title.TextSize = 18
    Title.Parent = Frame

    local Content = Instance.new("TextLabel")
    Content.Text = config.Content or "Mensagem"
    Content.Size = UDim2.new(1, 0, 0.6, 0)
    Content.Position = UDim2.new(0, 0, 0.4, 0)
    Content.BackgroundTransparency = 1
    Content.TextColor3 = LuqLib.Theme.SubText
    Content.Font = Enum.Font.Gotham
    Content.TextSize = 14
    Content.TextWrapped = true
    Content.Parent = Frame

    task.delay(config.Time or 5, function()
        Notification:Destroy()
    end)
end

return LuqLib
